Model: two-levels (students nested in schools) with a level-one predictor (student SES), level-two predictor (private vs. public school), and an interaction between them. The effect of SES has a varying slope over school. Outcome is student math score.

  • Traditional
  • Distributional
\[\begin{aligned} \text{MATH}_{ij} &= \beta_{0j} + \beta_{1j} \text{SES}_{ij} + r_{ij} \\ \beta_{0j} &= \gamma_{00} + \gamma_{01} \text{PRIVATE}_{0j} + u_{0j} \\ \beta_{1j} &= \gamma_{10} + \gamma_{11} \text{PRIVATE}_{1j} + u_{1j} \end{aligned}\]
\[\begin{aligned} \text{MATH}_{ij} &= \gamma_{00} + \gamma_{01} \text{PRIVATE}_{0j} + u_{0j} + (\gamma_{10} + \gamma_{11} \text{PRIVATE}_{1j} + u_{1j}) \text{SES}_{ij} + r_{ij} \\ \text{MATH}_{ij} &= \gamma_{00} + \gamma_{01} \text{PRIVATE}_{0j} + \gamma_{10} \text{SES}_{ij} + \gamma_{11} (\text{PRIVATE}_{1j} \times \text{SES}_{ij}) + u_{0j} + \text{SES}_{ij} u_{1j} + r_{ij} \end{aligned}\]
\[\begin{aligned} \text{MATH}_{i} &\sim Normal(\mu_{i}, \sigma) \\ \mu_i &= \mu_{school[i]} + \beta_{SES_{school[i]}} SES_i \\ \mu_{school[i]} &= \alpha_{school[i]} + (\beta_{PRI} + \beta_{PRI \times SES} SES_i) PRI_{school[i]} \\ \begin{bmatrix} \alpha_{school[i]}\\ \beta_{SES_{school[i]}} \end{bmatrix} &\sim MVN \left( \begin{bmatrix} \alpha \\ \beta_{\text{SES}} \end{bmatrix}, \textbf{S} \right) \\ \textbf{S} &= \begin{bmatrix} \sigma_{\alpha} & 0 \\ 0 & \sigma_{\beta_{\text{SES}}} \end{bmatrix} \textbf{R} \begin{bmatrix} \sigma_{\alpha} & 0 \\ 0 & \sigma_{\beta_{\text{SES}}} \end{bmatrix} \\[1em] \alpha &\sim Normal(0, 2) \\ \beta_{\text{SES}} &\sim Normal(0, 1) \\ \beta_{\text{PRI}} &\sim Normal(0, 1) \\ \beta_{\text{PRI} \times \text{SES}} &\sim Normal(0, 1) \\ (\sigma, \ \sigma_{\alpha}, \ \sigma_{\beta_{\text{SES}}}) &\sim Exp(1) \\ \textbf{R} &\sim LKJCorr(2) \end{aligned}\]

gen_data <- function(data_gen_params) {
  tibble(schoolid = seq(1, data_gen_params$param_n_schools),
         n_students = data_gen_params$param_n_students,
         private = rbinom(data_gen_params$param_n_schools, 1, .3),
         beta_ses = data_gen_params$ses,
         beta_private = data_gen_params$private,
         beta_private_by_ses = data_gen_params$`ses:private`) |>
    bind_cols(varying_effects = MASS::mvrnorm(data_gen_params$param_n_schools, 
                                              mu = c(data_gen_params$`(Intercept)`, data_gen_params$ses), 
                                              Sigma = data_gen_vcov) |>
                as_tibble() |>
                set_names(c("alpha_school", "beta_ses_school"))) |>
    uncount(n_students) |>
    mutate(studentid = row_number(),
           .after = schoolid) |>
    mutate(ses = rnorm(data_gen_params$param_n_schools * data_gen_params$param_n_students),
           private_by_ses = private * ses,
           mu_school = alpha_school + 
             beta_private * private +
             beta_private_by_ses * private_by_ses,
           mu = mu_school + beta_ses * ses,
           mathscore = rnorm(data_gen_params$param_n_schools * data_gen_params$param_n_students,
                        mean = mu, sd = data_gen_params$sd__Observation)) |>
    nest(gen_params = c(starts_with("beta"), starts_with("alpha"), starts_with("mu")))
}
